#include <stdio.h>

int sanitize_input(char *input) {
    for (int i = 0; i < 100; i++) {
        if (input[i] == '\n') {
            input[i] = '\0';
        }
    }
    return 0;
}

int main() {
    char input[100];
    fgets(input, sizeof(input), stdin);
    sanitize_input(input);
    printf(input);  
    return 0;
}
